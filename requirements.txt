import aiosqlite
from datetime import datetime

DB_PATH = 'data/bot.db'

CREATE_SQL = '''
CREATE TABLE IF NOT EXISTS users (
    user_id INTEGER PRIMARY KEY,
    username TEXT,
    is_vip INTEGER DEFAULT 0,
    created_at TEXT NOT NULL
);
CREATE TABLE IF NOT EXISTS signals (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    symbol TEXT NOT NULL,
    side TEXT NOT NULL,
    score INTEGER NOT NULL,
    price REAL,
    created_at TEXT NOT NULL
);
'''

async def init_db() -> None:
    async with aiosqlite.connect(DB_PATH) as db:
        await db.executescript(CREATE_SQL)
        await db.commit()

async def upsert_user(user_id: int, username: str | None) -> None:
    async with aiosqlite.connect(DB_PATH) as db:
        await db.execute(
            'INSERT OR IGNORE INTO users(user_id, username, created_at) VALUES(?, ?, ?)',
            (user_id, username, datetime.utcnow().isoformat())
        )
        await db.execute('UPDATE users SET username=? WHERE user_id=?', (username, user_id))
        await db.commit()

async def set_vip(user_id: int, value: bool) -> None:
    async with aiosqlite.connect(DB_PATH) as db:
        await db.execute(
            'INSERT OR IGNORE INTO users(user_id, created_at) VALUES(?, ?)',
            (user_id, datetime.utcnow().isoformat())
        )
        await db.execute('UPDATE users SET is_vip=? WHERE user_id=?', (1 if value else 0, user_id))
        await db.commit()

async def is_vip(user_id: int) -> bool:
    async with aiosqlite.connect(DB_PATH) as db:
        cur = await db.execute('SELECT is_vip FROM users WHERE user_id=?', (user_id,))
        row = await cur.fetchone()
        return bool(row and row[0])

async def save_signal(symbol: str, side: str, score: int, price: float) -> None:
    async with aiosqlite.connect(DB_PATH) as db:
        await db.execute(
            'INSERT INTO signals(symbol, side, score, price, created_at) VALUES(?, ?, ?, ?, ?)',
            (symbol, side, score, price, datetime.utcnow().isoformat())
        )
        await db.commit()
